def formatHGVSvep(element):
    # input = "c.960T>G;c.951T>G;c.1062T>G"
    # output = 'T/G'
    bun = re.findall(r"c.\d+(\D>\D)",element)
    str1 = ''.join(bun[0])
    fin = str1.replace('>','/')
    return fin
